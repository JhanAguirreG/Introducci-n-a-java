        System.out.println("División: " + division);
    }

Haz una aplicación que calcule el área de un círculo(pi*R2). El radio se pedirá por teclado (recuerda pasar de String a double con Double.parseDouble). Usa la constante PI y el método pow de Math.

import java.util.Scanner;

public class CalcularAreaCirculo {
    public static void main(String[] args) {
        // Crear un objeto Scanner para leer la entrada del usuario
        Scanner scanner = new Scanner(System.in);

        // Pedir al usuario que ingrese el radio del círculo
        System.out.print("Por favor, ingresa el radio del círculo: ");
        String radioStr = scanner.nextLine();

        // Convertir el radio de String a double
        double radio = Double.parseDouble(radioStr);

        // Calcular el área del círculo
        double area = Math.PI * Math.pow(radio, 2);

        // Mostrar el resultado por consola
        System.out.println("El área del círculo con radio " + radio + " es: " + area);
    }
}

Pedir un número entre 0 a 9999 y mostrarlo con las cifras al revés.

import java.util.Scanner;

public class NumeroAlReves {
    public static void main(String[] args) {
        // Crear un objeto Scanner para leer la entrada del usuario
        Scanner scanner = new Scanner(System.in);

        // Pedir al usuario que ingrese un número entre 0 y 9999
        System.out.print("Por favor, ingresa un número entre 0 y 9999: ");
        int numero = scanner.nextInt();

        // Verificar si el número está en el rango permitido
        if (numero >= 0 && numero <= 9999) {
            // Convertir el número a cadena para facilitar el reverso
            String numeroCadena = String.valueOf(numero);

            // Mostrar el número al revés
            System.out.print("El número " + numero + " al revés es: ");
            for (int i = numeroCadena.length() - 1; i >= 0; i--) {
                System.out.print(numeroCadena.charAt(i));
            }
        } else {
            System.out.println("El número ingresado está fuera del rango permitido.");
        }
    }
}

Dadas las edades y alturas de n estudiantes, mostrar la edad y la estatura media, la cantidad de alumnos mayores de 18 años, y la cantidad de estudiantes que miden más de 175 cm.

import java.util.Scanner;

public class EstadisticasEstudiantes {
    public static void main(String[] args) {
        // Crear un objeto Scanner para leer la entrada del usuario
        Scanner scanner = new Scanner(System.in);

        // Pedir al usuario que ingrese la cantidad de estudiantes
        System.out.print("Por favor, ingresa la cantidad de estudiantes: ");
        int cantidadEstudiantes = scanner.nextInt();

        // Variables para almacenar la suma de edades y alturas
        int sumaEdades = 0;
        double sumaAlturas = 0.0;

        // Variables para contar estudiantes mayores de 18 años y con altura mayor a 175 cm
        int estudiantesMayoresDe18 = 0;
        int estudiantesAlturaMayor175 = 0;

        // Leer las edades y alturas de los estudiantes
        for (int i = 1; i <= cantidadEstudiantes; i++) {
            System.out.println("Estudiante " + i + ":");
            System.out.print("Edad: ");
            int edad = scanner.nextInt();
            sumaEdades += edad;

            System.out.print("Altura (en cm): ");
            double altura = scanner.nextDouble();
            sumaAlturas += altura;

            if (edad > 18) {
                estudiantesMayoresDe18++;
            }

            if (altura > 175) {
                estudiantesAlturaMayor175++;
            }
        }

        // Calcular la media de edades y alturas
        double mediaEdades = (double) sumaEdades / cantidadEstudiantes;
        double mediaAlturas = sumaAlturas / cantidadEstudiantes;

        // Mostrar los resultados
        System.out.println("\nResultados:");
        System.out.println("Edad media de los estudiantes: " + mediaEdades);
        System.out.println("Altura media de los estudiantes (en cm): " + mediaAlturas);
        System.out.println("Cantidad de estudiantes mayores de 18 años: " + estudiantesMayoresDe18);
        System.out.println("Cantidad de estudiantes con altura mayor a 175 cm: " + estudiantesAlturaMayor175);
    }
}
